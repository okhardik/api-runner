<script>
const $ = (sel) => document.querySelector(sel);
const API = '';

// ── State ──────────────────────────────────────────────
let curlStr = '';
let parsedCurl = null;
let variables = [];
let dataRows = [];
let currentRunId = null;
let pollInterval = null;

// ── Curl parsing ───────────────────────────────────────
const curlInput = $('#curl-input');
let parseTimeout;
curlInput.addEventListener('input', () => {
  clearTimeout(parseTimeout);
  parseTimeout = setTimeout(parseCurl, 400);
});

async function parseCurl() {
  curlStr = curlInput.value.trim();
  if (!curlStr) {
    $('#parsed-info').classList.add('hidden');
    $('#var-badges').innerHTML = '';
    updateButtons();
    return;
  }
  try {
    const res = await fetch(`${API}/api/parse-curl`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ curl: curlStr }),
    });
    const data = await res.json();
    parsedCurl = data.parsed;
    variables = data.variables;

    $('#parsed-info').classList.remove('hidden');
    $('#parsed-info').innerHTML = `
      <span class="parsed-chip method">${parsedCurl.method || 'GET'}</span>
      <span class="parsed-chip">${parsedCurl.url.length > 70 ? parsedCurl.url.slice(0, 70) + '...' : parsedCurl.url}</span>
      <span class="parsed-chip">${Object.keys(parsedCurl.headers || {}).length} headers</span>
      ${parsedCurl.data ? '<span class="parsed-chip">has body</span>' : ''}
    `;
    $('#var-badges').innerHTML = variables.map(v =>
      `<span class="var-badge">{{${v}}}</span>`
    ).join('');
    updateButtons();
  } catch (e) {
    console.error(e);
  }
}

// ── File upload ────────────────────────────────────────
const uploadZone = $('#upload-zone');
const fileInput = $('#file-input');

uploadZone.addEventListener('click', () => fileInput.click());
uploadZone.addEventListener('dragover', (e) => { e.preventDefault(); uploadZone.classList.add('drag-over'); });
uploadZone.addEventListener('dragleave', () => uploadZone.classList.remove('drag-over'));
uploadZone.addEventListener('drop', (e) => {
  e.preventDefault();
  uploadZone.classList.remove('drag-over');
  if (e.dataTransfer.files.length) handleFile(e.dataTransfer.files[0]);
});
fileInput.addEventListener('change', () => {
  if (fileInput.files.length) handleFile(fileInput.files[0]);
});
$('#file-remove').addEventListener('click', clearFile);

async function handleFile(file) {
  const text = await file.text();
  try {
    const res = await fetch(`${API}/api/parse-data`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ content: text, filename: file.name }),
    });
    const data = await res.json();
    if (data.error) { alert('Parse error: ' + data.error); return; }

    dataRows = data.rows;
    uploadZone.classList.add('hidden');
    $('#file-loaded').classList.remove('hidden');
    $('#file-name').textContent = file.name;
    $('#file-meta').textContent = `${data.count} rows · ${data.columns.length} columns`;

    renderTable(data.columns, dataRows);
    updateButtons();
  } catch (e) {
    alert('Error reading file: ' + e.message);
  }
}

function clearFile() {
  dataRows = [];
  fileInput.value = '';
  uploadZone.classList.remove('hidden');
  $('#file-loaded').classList.add('hidden');
  $('#data-table-wrap').classList.add('hidden');
  updateButtons();
}

function renderTable(columns, rows) {
  const wrap = $('#data-table-wrap');
  const displayRows = rows.slice(0, 50);
  let html = '<table class="data-table"><thead><tr>';
  html += '<th class="row-num">#</th>';
  columns.forEach(c => { html += `<th>${esc(c)}</th>`; });
  html += '</tr></thead><tbody>';
  displayRows.forEach((row, i) => {
    html += '<tr>';
    html += `<td class="row-num">${i + 1}</td>`;
    columns.forEach(c => { html += `<td title="${esc(row[c] || '')}">${esc(row[c] || '')}</td>`; });
    html += '</tr>';
  });
  if (rows.length > 50) {
    html += `<tr><td colspan="${columns.length + 1}" style="text-align:center;color:var(--text2);padding:12px">... and ${rows.length - 50} more rows</td></tr>`;
  }
  html += '</tbody></table>';
  wrap.innerHTML = html;
  wrap.classList.remove('hidden');
}

function esc(s) {
  if (s == null) return '';
  const d = document.createElement('div');
  d.textContent = s;
  return d.innerHTML;
}

// ── Buttons ────────────────────────────────────────────
function updateButtons() {
  const ready = curlStr && parsedCurl && dataRows.length > 0;
  $('#btn-preview').disabled = !ready;
  $('#btn-run').disabled = !ready;
}

// ── Preview (Dry Run) ──────────────────────────────────
$('#btn-preview').addEventListener('click', async (e) => {
  const btn = e.target;
  const originalText = btn.innerHTML;
  
  try {
    btn.innerHTML = 'Loading...';
    btn.disabled = true;

    const sliced = getSlicedRows(); // Uses ALL rows according to your start/limit filters
    const res = await fetch(`${API}/api/dry-run`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ curl: curlStr, rows: sliced }),
    });

    if (!res.ok) {
      const errTxt = await res.text();
      throw new Error(`Server returned ${res.status}: ${errTxt}`);
    }

    const data = await res.json();
    if (data.error) throw new Error(data.error);

    let html = '';
    $('#preview-title').textContent = `Dry Run Preview (${data.previews.length} rows)`;

    data.previews.forEach((p, i) => {
      let bodyFormatted = p.body || '';
      try { bodyFormatted = JSON.stringify(JSON.parse(p.body), null, 2); } catch {}
      
      const headersStr = (p.headers && Object.keys(p.headers).length > 0)
        ? Object.entries(p.headers).map(([k,v]) => `${k}: ${v}`).join('\n') 
        : 'No headers';
      
      html += `
        <div class="preview-item">
          <div class="p-header">[${i + 1}/${data.previews.length}] ${p.method || 'GET'} ${p.url || ''}</div>
          <div class="section-label">HEADERS</div>
          <pre>${esc(headersStr)}</pre>
          ${bodyFormatted ? `<div class="section-label">BODY</div><pre>${esc(bodyFormatted)}</pre>` : ''}
        </div>`;
    });
    
    $('#preview-content').innerHTML = html;
    $('#preview-modal').classList.add('show');
    
  } catch (err) {
    alert("Preview failed: " + err.message);
    console.error(err);
  } finally {
    btn.innerHTML = originalText;
    updateButtons();
  }
});

// ── Run ────────────────────────────────────────────────
$('#btn-run').addEventListener('click', async () => {
  const sliced = getSlicedRows();
  if (!confirm(`Send ${sliced.length} API requests?`)) return;

  const delay = parseInt($('#delay-input').value) || 200;
  const timeout = parseInt($('#timeout-input').value) || 30;

  const res = await fetch(`${API}/api/run`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ curl: curlStr, rows: sliced, delay, timeout }),
  });
  const data = await res.json();
  currentRunId = data.run_id;

  $('#step-results').classList.remove('hidden');
  $('#btn-abort').classList.remove('hidden');
  $('#btn-run').disabled = true;
  $('#results-list').innerHTML = '';
  $('#progress-bar').style.width = '0%';
  $('#progress-bar').className = 'progress-bar';

  let lastResultCount = 0;
  pollInterval = setInterval(async () => {
    try {
      const r = await fetch(`${API}/api/run/${currentRunId}`);
      const run = await r.json();
      updateRunUI(run, lastResultCount);
      lastResultCount = run.results.length;

      if (run.status === 'completed' || run.status === 'aborted') {
        clearInterval(pollInterval);
        $('#btn-abort').classList.add('hidden');
        $('#btn-run').disabled = false;
        $('#progress-bar').classList.add(run.fail_count > 0 ? 'error' : 'done');
      }
    } catch (e) { console.error(e); }
  }, 500);
});

$('#btn-abort').addEventListener('click', async () => {
  if (currentRunId) {
    await fetch(`${API}/api/run/${currentRunId}/abort`, { method: 'POST' });
  }
});

function updateRunUI(run, prevCount) {
  const total = run.total;
  const done = run.results.length;
  const pct = total > 0 ? (done / total * 100).toFixed(1) : 0;
  $('#progress-bar').style.width = pct + '%';

  $('#run-stats').innerHTML = `
    <span class="stat"><span class="dot yellow"></span> ${done}/${total} processed</span>
    <span class="stat"><span class="dot green"></span> ${run.success_count} success</span>
    <span class="stat"><span class="dot red"></span> ${run.fail_count} failed</span>
    <span class="stat" style="color:var(--text2)">${run.status}</span>
  `;

  const list = $('#results-list');
  for (let i = prevCount; i < run.results.length; i++) {
    const r = run.results[i];
    list.appendChild(createResultItem(r, i, total));
  }
  list.scrollTop = list.scrollHeight;
}

function createResultItem(r, index, total) {
  const div = document.createElement('div');
  div.className = 'result-item';

  const statusCode = r.error ? 'ERR' : r.status;
  const statusClass = r.error ? 'err'
    : r.status < 300 ? 's2xx'
    : r.status < 500 ? 's4xx' : 's5xx';

  const varsStr = Object.entries(r.variables || {}).slice(0, 3).map(([k,v]) => `${k}=${v}`).join(' | ');

  let bodyStr = '';
  if (r.body) {
    try { bodyStr = typeof r.body === 'string' ? r.body : JSON.stringify(r.body, null, 2); }
    catch { bodyStr = String(r.body); }
  }

  let reqBodyStr = '';
  if (r.request && r.request.body) {
    try { reqBodyStr = JSON.stringify(JSON.parse(r.request.body), null, 2); }
    catch { reqBodyStr = r.request.body; }
  }
  
  const headersStr = (r.request && r.request.headers && Object.keys(r.request.headers).length > 0)
    ? Object.entries(r.request.headers).map(([k,v]) => `${k}: ${v}`).join('\n')
    : 'No headers';

  div.innerHTML = `
    <div class="result-header" onclick="this.parentElement.classList.toggle('open')">
      <span class="result-index">#${index + 1}</span>
      <span class="result-status ${statusClass}">${statusCode}</span>
      <span class="result-time">${r.time_ms}ms</span>
      <span class="result-vars">${esc(varsStr)}</span>
      <span class="result-chevron">&#9654;</span>
    </div>
    <div class="result-body">
      <div class="section-label">Request Info</div>
      <pre>${r.request ? esc((r.request.method || 'GET') + ' ' + (r.request.url || '')) : ''}</pre>
      <div class="section-label">Request Headers</div>
      <pre>${esc(headersStr)}</pre>
      ${reqBodyStr ? `<div class="section-label">Request Body</div><pre>${esc(reqBodyStr)}</pre>` : ''}
      <div class="section-label">Response${r.error ? ' Error' : ''}</div>
      <pre>${esc(r.error || bodyStr)}</pre>
    </div>
  `;
  return div;
}

  a.download = `api_runner_${currentRunId}.json`;
  a.click();
});

// ── Helpers ────────────────────────────────────────────
function getSlicedRows() {
  let rows = [...dataRows];
  const start = parseInt($('#start-input').value) || 0;
  const limit = parseInt($('#limit-input').value) || 0;
  if (start > 0) rows = rows.slice(start);
  if (limit > 0) rows = rows.slice(0, limit);
  return rows;
}
