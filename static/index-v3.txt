<script>
const $ = s => document.querySelector(s);
let dataRows = [];
let currentRunId = null;

// Logic
$('#curl-input').oninput = async () => {
  const res = await fetch('/api/parse-curl', {
    method: 'POST', headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({curl: $('#curl-input').value})
  });
  const data = await res.json();
  $('#var-badges').innerHTML = (data.variables || []).map(v => `<span class="var-badge">{{${v}}}</span>`).join('');
  checkReady();
};

$('#upload-zone').onclick = () => $('#file-input').click();
$('#file-input').onchange = async (e) => {
  const file = e.target.files[0];
  if(!file) return;
  const text = await file.text();
  const res = await fetch('/api/parse-data', {
    method: 'POST', headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({content: text, filename: file.name})
  });
  const data = await res.json();
  dataRows = data.rows;
  $('#upload-zone').innerText = `File: ${file.name} (${data.count} rows)`;
  renderTable(data.columns, data.rows.slice(0, 5));
  checkReady();
};

function renderTable(cols, rows) {
  let html = `<table><thead><tr>${cols.map(c=>`<th>${c}</th>`).join('')}</tr></thead><tbody>`;
  rows.forEach(r => { html += `<tr>${cols.map(c=>`<td>${r[c]||''}</td>`).join('')}</tr>`; });
  html += `</tbody></table>`;
  $('#data-table-wrap').innerHTML = html;
  $('#data-table-wrap').classList.remove('hidden');
}

function checkReady() {
  const ready = $('#curl-input').value && dataRows.length;
  $('#run-btn').disabled = !ready;
  $('#preview-btn').disabled = !ready;
}

function getSlicedRows() {
  let rows = [...dataRows];
  const start = parseInt($('#start-at').value) || 0;
  const limit = parseInt($('#limit').value) || 0;
  if(start > 0) rows = rows.slice(start);
  if(limit > 0) rows = rows.slice(0, limit);
  return rows;
}

// Preview
$('#preview-btn').onclick = async () => {
  const rows = getSlicedRows().slice(0, 5);
  const res = await fetch('/api/dry-run', {
    method: 'POST', headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({curl: $('#curl-input').value, rows})
  });
  const data = await res.json();
  $('#preview-content').innerHTML = data.previews.map((p, i) => `
    <div style="margin-bottom:20px; padding:15px; background:var(--surface2); border-radius:8px">
      <div style="color:var(--accent); font-weight:bold; margin-bottom:10px">#${i+1} ${p.method} ${p.url}</div>
      <div class="section-label">Headers</div>
      <pre>${JSON.stringify(p.headers, null, 2)}</pre>
      ${p.body ? `<div class="section-label">Body</div><pre>${p.body}</pre>` : ''}
    </div>
  `).join('');
  $('#modal').style.display = 'flex';
};

// Run
$('#run-btn').onclick = async () => {
  const sliced = getSlicedRows();
  if(!confirm(`Run ${sliced.length} requests?`)) return;
  
  const res = await fetch('/api/run', {
    method: 'POST', headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({
      curl: $('#curl-input').value, 
      rows: sliced, 
      delay: parseInt($('#delay').value),
      timeout: parseInt($('#timeout').value)
    })
  });
  const {run_id} = await res.json();
  currentRunId = run_id;
  
  $('#results-step').classList.remove('hidden');
  $('#abort-btn').classList.remove('hidden');
  $('#results-list').innerHTML = '';
  
  const poll = setInterval(async () => {
    const r = await fetch(`/api/run/${run_id}`);
    const run = await r.json();
    const pct = (run.results.length / run.total) * 100;
    $('#progress-bar').style.width = pct + '%';
    $('#stats').innerText = `Progress: ${run.results.length}/${run.total} | Success: ${run.success_count} | Failed: ${run.fail_count} | Status: ${run.status}`;
    
    const currentCount = document.querySelectorAll('.result-item').length;
    for(let i = currentCount; i < run.results.length; i++) {
      $('#results-list').prepend(createItem(run.results[i], i));
    }
    
    if (run.status === 'completed' || run.status === 'aborted') {
      clearInterval(poll);
      $('#abort-btn').classList.add('hidden');
    }
  }, 1000);
};

$('#abort-btn').onclick = () => fetch(`/api/run/${currentRunId}/abort`, {method:'POST'});

$('#export-btn').onclick = async () => {
    const r = await fetch(`/api/run/${currentRunId}`);
    const data = await r.json();
    const blob = new Blob([JSON.stringify(data.results, null, 2)], {type:'application/json'});
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url; a.download = `run_${currentRunId}.json`; a.click();
};

function createItem(res, idx) {
  const div = document.createElement('div');
  div.className = 'result-item';
  const status = res.error ? 'ERR' : (res.status || '???');
  const statusClass = res.error ? 'err' : (res.status < 300 ? 's2xx' : (res.status < 500 ? 's4xx' : 's5xx'));
  const body = typeof res.body === 'object' ? JSON.stringify(res.body, null, 2) : (res.body || '');

  div.innerHTML = `
    <div class="result-header" onclick="this.parentElement.classList.toggle('open')">
      <span class="chevron">▶</span>
      <span style="color:var(--text2); min-width:30px">#${idx+1}</span>
      <span class="result-status ${statusClass}">${status}</span>
      <span style="flex:1; overflow:hidden; text-overflow:ellipsis; white-space:nowrap; font-family:var(--mono)">${res.request.url}</span>
      <span style="font-size:11px; color:var(--text2)">${res.time_ms}ms</span>
    </div>
    <div class="result-body">
      <div class="section-label">Request Info</div>
      <pre>${res.request.method} ${res.request.url}\n\nHeaders:\n${JSON.stringify(res.request.headers, null, 2)}</pre>
      ${res.request.body ? `<div class="section-label">Request Body</div><pre>${res.request.body}</pre>` : ''}
      <div class="section-label">Response Content</div>
      <pre>${res.error || body || 'Empty Response'}</pre>
    </div>
  `;
  return div;
}
</script>
