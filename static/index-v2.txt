<script>
const $ = s => document.querySelector(s);
let dataRows = [];

// Parse Curl
$('#curl-input').oninput = async () => {
  const res = await fetch('/api/parse-curl', {
    method: 'POST', headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({curl: $('#curl-input').value})
  });
  const data = await res.json();
  $('#var-badges').innerHTML = (data.variables || []).map(v => `<span class="var-badge">{{${v}}}</span>`).join('');
  checkReady();
};

// Handle File
$('#upload-zone').onclick = () => $('#file-input').click();
$('#file-input').onchange = async (e) => {
  const file = e.target.files[0];
  const text = await file.text();
  const res = await fetch('/api/parse-data', {
    method: 'POST', headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({content: text, filename: file.name})
  });
  const data = await res.json();
  dataRows = data.rows;
  $('#upload-zone').innerText = `Loaded ${data.count} rows from ${file.name}`;
  renderTable(data.columns, data.rows.slice(0, 5));
  checkReady();
};

function renderTable(cols, rows) {
  let html = `<table><thead><tr>${cols.map(c=>`<th>${c}</th>`).join('')}</tr></thead><tbody>`;
  rows.forEach(r => { html += `<tr>${cols.map(c=>`<td>${r[c]}</td>`).join('')}</tr>`; });
  html += `</tbody></table>`;
  $('#data-table-wrap').innerHTML = html;
  $('#data-table-wrap').classList.remove('hidden');
}

function checkReady() {
  $('#run-btn').disabled = !($('#curl-input').value && dataRows.length);
}

// Run
$('#run-btn').onclick = async () => {
  const res = await fetch('/api/run', {
    method: 'POST', headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({curl: $('#curl-input').value, rows: dataRows, delay: parseInt($('#delay').value)})
  });
  const {run_id} = await res.json();
  $('#results-step').classList.remove('hidden');
  $('#results-list').innerHTML = '';
  
  const poll = setInterval(async () => {
    const r = await fetch(`/api/run/${run_id}`);
    const run = await r.json();
    
    const pct = (run.results.length / run.total) * 100;
    $('#progress-bar').style.width = pct + '%';
    $('#stats').innerText = `Processed: ${run.results.length} / ${run.total} | Success: ${run.success_count} | Failed: ${run.fail_count}`;
    
    // Simple append for new results
    const currentCount = document.querySelectorAll('.result-item').length;
    for(let i = currentCount; i < run.results.length; i++) {
      $('#results-list').prepend(createItem(run.results[i], i));
    }
    
    if (run.status === 'completed' || run.status === 'aborted') clearInterval(poll);
  }, 1000);
};

function createItem(res, idx) {
  const div = document.createElement('div');
  div.className = 'result-item';
  const status = res.error ? 'ERR' : res.status;
  const statusClass = res.error ? 'err' : (res.status < 300 ? 's2xx' : 's4xx');
  const body = typeof res.body === 'object' ? JSON.stringify(res.body, null, 2) : res.body;

  div.innerHTML = `
    <div class="result-header" onclick="this.parentElement.classList.toggle('open')">
      <span style="color:var(--text2)">#${idx+1}</span>
      <span class="result-status ${statusClass}">${status}</span>
      <span style="flex:1; overflow:hidden; text-overflow:ellipsis; white-space:nowrap;">${res.request.url}</span>
      <span>${res.time_ms}ms</span>
    </div>
    <div class="result-body">
      <div class="section-label">Request Body</div>
      <pre>${res.request.body || 'None'}</pre>
      <div class="section-label">Response Content</div>
      <pre>${res.error || body || 'Empty Response'}</pre>
    </div>
  `;
  return div;
}
